public class TabAdapterHome extends PagerAdapter {

    }

    @Override
    public Object instantiateItem(@NonNull ViewGroup container, int position) {
        View view;
        view =  LayoutInflater.from(getContext()).inflate(R.layout.layout_trang_chu
                container, false);
        container.addView(view);
        return view;
        return null;
    }

    public int getCount() {

    @Override
    public boolean isViewFromObject(@NonNull View view, @NonNull Object object) {
